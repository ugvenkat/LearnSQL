CREATE OR REPLACE PROCEDURE GetStudentName (p_id IN NUMBER, p_Student_name OUT VARCHAR)
IS
BEGIN
SELECT StudentName INTO p_Student_name FROM student WHERE StudentId = p_id;
END;
/

DECLARE 
l_student_Name varchar(50);
BEGIN
GetStudentName(1,l_student_Name);
dbms_output.put_line(l_student_Name);
END;
/


create or replace procedure get_area
(n_length in number,
 n_width in number,
 n_area out number)
 as
 begin
 n_area := n_length*n_width;
 end get_area;


 declare
 n_area number :=0;
 begin
 get_area(10,20,n_area);
 dbms_output.put_line('Area '||n_area);
 end;

CREATE OR REPLACE FUNCTION fn_GetStudentName
(p_studentId Number)
RETURN VARCHAR
IS
student_name VARCHAR(50); 
BEGIN 
SELECT StudentName INTO student_name FROM student WHERE StudentId = p_studentId;
RETURN student_name;
END;
/

CREATE OR REPLACE FUNCTION f_to_c 
(farenheit NUMBER) 
RETURN NUMBER 
IS
centigrade NUMBER;
BEGIN
centigrade := (farenheit - 32) * 5/9; 
RETURN centigrade;
END;
/

select f_to_c(100) from dual;
